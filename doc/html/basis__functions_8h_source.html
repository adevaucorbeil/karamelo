40  inline double cubic_spline(double r, int ntype)
41  {
42  if (r >= 1 && r < 2) {
43  if (ntype==1) {
44  return 0;
45  } else {
46  return -1.0/6.0*r*r*r + r*r - 2*r + 4.0/3.0;
47  }
48  } else if (r >=0 && r < 1) {
49  if (ntype==-2) {
50  return 1.0/6.0*r*r*r - r + 1;
51  } else if (ntype==2) {
52  return 1;
53  } else if (ntype==1) {
54  return 1.0/3.0*r*r*r - r*r + 2.0/3.0;
55  } else {
56  return 0.5*r*r*r - r*r + 2.0/3.0;
57  }
58  } else if (r >= -1 && r < 0) {
59  if (ntype==2) {
60  return -1.0/6.0*r*r*r + r + 1;
61  } else if (ntype==-1) {
62  return -1.0/3.0*r*r*r - r*r + 2.0/3.0;
63  } else {
64  return -0.5*r*r*r - r*r + 2.0/3.0;
65  }
66  } else if (r >= -2 && r < -1) {
67  return 1.0/6.0*r*r*r + r*r + 2*r + 4.0/3.0;
68  } else {
69  return 0;
70  }
71  }

73  inline double derivative_cubic_spline(double r, int ntype, double icellsize)
74  {
75  if (r >= 1 && r < 2) {
76  if (ntype==1) {
77  return -icellsize;// * (-1);
78  } else {
79  return icellsize * (-0.5*r*r + 2*r - 2);
80  }
81  } else if (r >=0 && r < 1) {
82  if (ntype==-2) {
83  return icellsize * (0.5*r*r - 1);
84  } else if (ntype==2) {
85  return icellsize;// * (1);
86  } else if (ntype==1) {
87  return icellsize * (r*r - 2*r);
88  } else {
89  return icellsize * (3.0/2.0*r*r - 2*r);
90  }
91  } else if (r >= -1 && r < 0) {
92  if (ntype==2) {
93  return icellsize * (-0.5*r*r + 1);
94  } else if (ntype==-1) {
95  return icellsize * (-r*r - 2*r);
96  } else {
97  return icellsize * (-3.0/2.0*r*r - 2*r);
98  }
99  } else if (r >= -2 && r < -1) {
100  return icellsize * (0.5*r*r + 2*r + 2);
101  } else {
102  return 0;
103  }
104  }

106  inline double bernstein_quadratic(double r_, int ntype)
107  {
108  double r = fabs(r_);
109  if (r >= 1.0) return 0;
110 
111  if (ntype==1) {
112  // Inside node:
113  if (r >= 0.5) return 0;
114  else return 0.5-2*r*r;
115  } else {
116  // Edge node:
117  return (1-r)*(1-r);
118  }
119  }

125
